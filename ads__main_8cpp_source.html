16bool ADS_Eclipse_Decision(const ADCS_CALC_TYPE luxData[ADCS_NUM_PANELS], const ADCS_CALC_TYPE threshold) {
17
18 ADCS_CALC_TYPE sum = 0;
19 // skip luxData[0] - not solar cell power
20 for(uint8_t i = 1; i < ADCS_NUM_PANELS; i++) {
21 sum += luxData[i];
22 }
23
24 if(sum < threshold) {
25 return(true);
26 }
27
28 return(false);
29}

32void ADS_Main(const ADCS_CALC_TYPE omega[ADCS_NUM_AXES], const ADCS_CALC_TYPE magData[ADCS_NUM_AXES], const ADCS_CALC_TYPE stateVars[ADCS_STATE_DIM],
33 const ADCS_CALC_TYPE controlVector[ADCS_NUM_AXES], ADCS_CALC_TYPE matrixP[ADCS_STATE_DIM][ADCS_STATE_DIM], const ADCS_CALC_TYPE solarEphe[ADCS_NUM_AXES],
34 const ADCS_CALC_TYPE magEphe[ADCS_NUM_AXES], ADCS_CALC_TYPE filtered_y[ADCS_STATE_DIM], ADCS_CALC_TYPE newAnglesVector[ADCS_NUM_AXES]) {
35
36 // Relevant variables declaration
37 ADCS_CALC_TYPE anglesInt[ADCS_NUM_AXES]; // Angular determination variables
38 ADCS_CALC_TYPE rotationMatrix[ADCS_NUM_AXES][ADCS_NUM_AXES]; // Rotation matrix
39
40 // Previous state of the system from t = t_1-1
41 ADCS_CALC_TYPE prevAngles[ADCS_NUM_AXES];
42 prevAngles[0] = stateVars[0];
43 prevAngles[1] = stateVars[1];
44 prevAngles[2] = stateVars[2];
45
46 // Measurements from sensors
47 ADCS_CALC_TYPE luxData[ADCS_NUM_PANELS];
48 luxData[0] = Sensors_Read_Light(lightSensorPanelY);
49 luxData[1] = Sensors_Current_ReadPower(currSensorXA);
50 luxData[2] = Sensors_Current_ReadPower(currSensorZA);
51 luxData[3] = Sensors_Current_ReadPower(currSensorY);
52 luxData[4] = Sensors_Current_ReadPower(currSensorXB);
53 luxData[5] = Sensors_Current_ReadPower(currSensorZB);
54 FOSSASAT_DEBUG_PRINT_ADCS_VECTOR(luxData, ADCS_NUM_PANELS);
55
56 // Decide whether the satellite is in eclipse situation or under light
57 if(ADS_Eclipse_Decision(luxData, adcsParams.eclipseThreshold)) {
58 // Generation of the measurements
59 FOSSASAT_DEBUG_PRINTLN(F("eclipse"));
60 ADS_Euler_Integrator(omega, prevAngles, anglesInt, adcsParams.timeStep);
61 ADS_Eclipse_Hybrid(magData, magEphe, rotationMatrix);
62
63 } else {
64 FOSSASAT_DEBUG_PRINTLN(F("no eclipse"));
65 ADCS_CALC_TYPE solarEphBody[ADCS_NUM_AXES]; // Solar ephemerides in the body frame
66
67 // Generation of the measurements
68 ADS_Euler_Integrator(omega, prevAngles, anglesInt, adcsParams.timeStep);
69 ADS_Solar_Determination(luxData, solarEphBody);
70 ADS_Measurement_Hybrid(solarEphe, magEphe, solarEphBody, magData, rotationMatrix);
71 }
72
73 // Processing of the measurements
74 ADS_Angles_Determination(rotationMatrix, newAnglesVector);
75 ADS_Rotation_Verification(newAnglesVector, anglesInt, adcsParams.rotationWeightRatio, adcsParams.rotationTrigger);
76
77 // Generation of the state point
78 ADCS_CALC_TYPE measurements[ADCS_STATE_DIM];
79 measurements[0] = newAnglesVector[0];
80 measurements[1] = newAnglesVector[1];
81 measurements[2] = newAnglesVector[2];
82 measurements[3] = omega[0];
83 measurements[4] = omega[1];
84 measurements[5] = omega[2];
85
86 // Filtering of the signal
87 ADS_Kalman_Filter(adcsParams.disturbCovariance, adcsParams.noiseCovariance, adcsParams.timeStep, stateVars, measurements, controlVector, adcsParams.inertiaTensor, matrixP, filtered_y);
88}
