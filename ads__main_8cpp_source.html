16 bool ADS_Eclipse_Decision(const ADCS_CALC_TYPE luxData[ADCS_NUM_PANELS], const ADCS_CALC_TYPE threshold) {
17 
18  ADCS_CALC_TYPE sum = 0;
19  // skip luxData[0] - not solar cell power
20  for(uint8_t i = 1; i < ADCS_NUM_PANELS; i++) {
