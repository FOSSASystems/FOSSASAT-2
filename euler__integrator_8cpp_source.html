14 void ADS_Euler_Integrator(const ADCS_CALC_TYPE omega[ADCS_NUM_AXES], const ADCS_CALC_TYPE currentAngles[ADCS_NUM_AXES],
15  ADCS_CALC_TYPE nextAngles[ADCS_NUM_AXES], const ADCS_CALC_TYPE delta_t) {
16  ADCS_CALC_TYPE p = omega[0];
17  ADCS_CALC_TYPE q = omega[1];
18  ADCS_CALC_TYPE r = omega[2];
19  ADCS_CALC_TYPE psi = currentAngles[0];
20  ADCS_CALC_TYPE phi = currentAngles[1];
21  ADCS_CALC_TYPE theta = currentAngles[2];
22 
23  // convert to SI units
24  ADCS_CALC_TYPE delta_t_sec = delta_t / 1000.0;
25 
26  // New Euler angles at time t+1
27  nextAngles[0] = psi + delta_t_sec*(p + (q*sin(psi) + r*cos(psi))*ADCS_Add_Tolerance(tan(phi), M_PI/2.0));
28  nextAngles[1] = phi + delta_t_sec*(q*cos(psi) - r*sin(psi));
29  nextAngles[2] = theta + delta_t_sec*((q*sin(psi) + r*cos(psi))*(1.0/(ADCS_Add_Tolerance(cos(phi), 0))));
30 }
