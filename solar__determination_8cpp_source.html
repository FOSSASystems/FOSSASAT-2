14 void ADS_Solar_Determination(const ADCS_CALC_TYPE luxData[ADCS_NUM_PANELS], ADCS_CALC_TYPE solarEph[ADCS_NUM_AXES],
15  ADCS_CALC_TYPE redundantSolarEph[ADCS_NUM_AXES]) {
16  // Constants definitions
17  ADCS_CALC_TYPE panelUnitVector[ADCS_NUM_PANELS][ADCS_NUM_AXES] = ADCS_PANEL_UNIT_VECTOR; // Unitary vector pointing to the solar panels in inverse form
18 
19  // Maximum obtainable power/lux array
20  float val;
21  ADCS_CALC_TYPE maxPower[ADCS_NUM_PANELS];
22  for(uint8_t i = 0; i < ADCS_NUM_PANELS - 1; i++) {
23  val = PersistentStorage_Get<float>(FLASH_STATS_POWER_XA + 2*(sizeof(float)) + i*3*(sizeof(float)));
24  maxPower[i] = (ADCS_CALC_TYPE)val;
25  }
26  val = PersistentStorage_Get<float>(FLASH_STATS_LIGHT_PANEL_Y + 2*(sizeof(float)));
27  maxPower[ADCS_NUM_PANELS - 1] = (ADCS_CALC_TYPE)val;
28 
29  // Main calculation
30  for(uint8_t i = 0; i < ADCS_NUM_AXES; i++) {
31  solarEph[i] = 0;
32  for(uint8_t j = 0; j < ADCS_NUM_AXES; j++) {
33  solarEph[i] += panelUnitVector[i][j]*luxData[j]*(1.0/maxPower[j]);
34  }
35  }
36 
37  for(uint8_t i = 0; i < ADCS_NUM_AXES; i++) {
38  redundantSolarEph[i] = 0;
39  for(uint8_t j = 0; j < ADCS_NUM_AXES; j++) {
40  redundantSolarEph[i] += panelUnitVector[ADCS_NUM_AXES + i][j]*luxData[ADCS_NUM_AXES + j]*(1.0/maxPower[ADCS_NUM_AXES + j]);
41  }
42  }
43 }
44 
45  // Normalize the solar ephemerides
46  ADCS_CALC_TYPE solarEphNorm = ADCS_VectorNorm(solarEph);
47  ADCS_CALC_TYPE redundantSolarEphNorm = ADCS_VectorNorm(redundantSolarEph);
48 
49  for(uint8_t i = 0; i < ADCS_NUM_AXES; i++) {
50  solarEph[i] /= ADCS_Add_Tolerance(solarEphNorm, 0.0);
51  redundantSolarEph[i] /= ADCS_Add_Tolerance(redundantSolarEphNorm, 0.0);
52  }
