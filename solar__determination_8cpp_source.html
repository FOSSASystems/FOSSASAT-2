14 void ADS_Solar_Determination(const ADCS_CALC_TYPE luxData[ADCS_NUM_PANELS], ADCS_CALC_TYPE solarEph[ADCS_NUM_AXES]) {
15  // Constants definitions
16  ADCS_CALC_TYPE panelUnitVector[ADCS_NUM_AXES][ADCS_NUM_PANELS] = ADCS_PANEL_UNIT_VECTOR; // Unitary vector pointing to the solar panels in pseudo-inverse form
17 
18  // Maximum obtainable power/lux array
19  ADCS_CALC_TYPE maxPower[ADCS_NUM_PANELS];
20  maxPower[0] = (ADCS_CALC_TYPE)PersistentStorage_Get<float>(FLASH_STATS_LIGHT_PANEL_Y + 2*(sizeof(float)));
21  maxPower[1] = (ADCS_CALC_TYPE)PersistentStorage_Get<float>(FLASH_STATS_POWER_XA) + 2*(sizeof(float));
22  maxPower[2] = (ADCS_CALC_TYPE)PersistentStorage_Get<float>(FLASH_STATS_POWER_ZA + 2*(sizeof(float)));
23  maxPower[3] = (ADCS_CALC_TYPE)PersistentStorage_Get<float>(FLASH_STATS_POWER_Y + 2*(sizeof(float)));
24  maxPower[4] = (ADCS_CALC_TYPE)PersistentStorage_Get<float>(FLASH_STATS_POWER_XB + 2*(sizeof(float)));
25  maxPower[5] = (ADCS_CALC_TYPE)PersistentStorage_Get<float>(FLASH_STATS_POWER_ZB + 2*(sizeof(float)));
26 
27  // Main calculation
28  for(uint8_t i = 0; i < ADCS_NUM_AXES; i++) {
29  solarEph[i] = 0;
30  for(uint8_t j = 0; j < ADCS_NUM_PANELS; j++) {
31  solarEph[i] += panelUnitVector[i][j]*(luxData[j]/maxPower[j]);
32  }
33  }
34 
35  // Normalize the solar ephemerides
36  ADCS_CALC_TYPE solarEphNorm = ADCS_Add_Tolerance(ADCS_VectorNorm(solarEph),0);
37 
38  for(uint8_t i = 0; i < ADCS_NUM_AXES; i++) {
39  solarEph[i] *= (1.0/solarEphNorm);
40  }
41 }
