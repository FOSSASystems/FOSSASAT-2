15 bool ADS_Rotation_Verification(ADCS_CALC_TYPE angles1[ADCS_NUM_AXES], const ADCS_CALC_TYPE angles2[ADCS_NUM_AXES],
16  const ADCS_CALC_TYPE weightRatio, const ADCS_CALC_TYPE trigger) {
17  // Constants definition
18  const ADCS_CALC_TYPE weight1 = weightRatio;
19  const ADCS_CALC_TYPE weight2 = 1.0 - weight1;
20 
21  // Compare both vectors and compute the norm
22  ADCS_CALC_TYPE diff[ADCS_NUM_AXES];
23  diff[0] = angles1[0] - angles2[0];
24  diff[1] = angles1[1] - angles2[1];
25  diff[2] = angles1[2] - angles2[2];
26 
27  // Arithmetic average of both measurements
28  if(ADCS_VectorNorm(diff) >= trigger) {
29  angles1[0] = weight1*angles1[0] + weight2*angles2[0];
30  angles1[1] = weight1*angles1[1] + weight2*angles2[1];
31  angles1[2] = weight1*angles1[2] + weight2*angles2[2];
32  return(false);
33  }
34 
35  return(true);
36 }
