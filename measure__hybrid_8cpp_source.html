14 void ADS_Measurement_Hybrid(const ADCS_CALC_TYPE v_1[ADCS_NUM_AXES], const ADCS_CALC_TYPE v_2[ADCS_NUM_AXES], const ADCS_CALC_TYPE v_3[ADCS_NUM_AXES],
15  const ADCS_CALC_TYPE m_1[ADCS_NUM_AXES], const ADCS_CALC_TYPE m_2[ADCS_NUM_AXES], const ADCS_CALC_TYPE m_3[ADCS_NUM_AXES],
16  ADCS_CALC_TYPE eulerAnglesMatrix[ADCS_NUM_AXES][ADCS_NUM_AXES]) {
17 
18  // Normalize both ephemeris and measurements vectors
19  ADCS_CALC_TYPE m1[ADCS_NUM_AXES] = m_1;
20  ADCS_CALC_TYPE m2[ADCS_NUM_AXES] = m_2;
21  ADCS_CALC_TYPE m3[ADCS_NUM_AXES] = m_3;
22  ADCS_CALC_TYPE v1[ADCS_NUM_AXES] = v_1;
23  ADCS_CALC_TYPE v2[ADCS_NUM_AXES] = v_2;
24  ADCS_CALC_TYPE v3[ADCS_NUM_AXES] = v_3;
25 
26  ADCS_CALC_TYPE m1_norm = ADCS_Add_Tolerance(ADCS_VectorNorm(m_1),0);
27  ADCS_CALC_TYPE m2_norm = ADCS_Add_Tolerance(ADCS_VectorNorm(m_2),0);
28  ADCS_CALC_TYPE m3_norm = ADCS_Add_Tolerance(ADCS_VectorNorm(m_3),0);
29 
30  ADCS_CALC_TYPE v1_norm = ADCS_Add_Tolerance(ADCS_VectorNorm(v_1),0);
31  ADCS_CALC_TYPE v2_norm = ADCS_Add_Tolerance(ADCS_VectorNorm(v_2),0);
32  ADCS_CALC_TYPE v3_norm = ADCS_Add_Tolerance(ADCS_VectorNorm(v_3),0);
33 
34  for(uint8_t i = 0; i < ADCS_NUM_AXES; i++) {
35  m1[i] *= (1.0/m1_norm);
36  m2[i] *= (1.0/m2_norm);
37  m3[i] *= (1.0/m3_norm);
38 
39  v1[i] *= (1.0/v1_norm);
40  v2[i] *= (1.0/v2_norm);
41  v3[i] *= (1.0/v3_norm);
42  }
43 
44  // Measurements matrix
45  ADCS_CALC_TYPE measMaxtrix[ADCS_NUM_AXES][ADCS_NUM_AXES];
46  measMaxtrix[0][0] = m1[0];
47  measMaxtrix[0][1] = m2[0];
48  measMaxtrix[0][2] = m3[0];
49  measMaxtrix[1][0] = m1[1];
50  measMaxtrix[1][1] = m2[1];
51  measMaxtrix[1][2] = m3[1];
52  measMaxtrix[2][0] = m1[2];
53  measMaxtrix[2][1] = m2[2];
54  measMaxtrix[2][2] = m3[2];
55 
56  // Inverse theoretical matrix
57  ADCS_CALC_TYPE invV[ADCS_NUM_AXES][ADCS_NUM_AXES];
58  const ADCS_CALC_TYPE det_V = ( (v1[0]*v2[1]*v3[2]) - (v1[2]*v2[1]*v3[0]) + (v1[1]*v2[2]*v3[0])
59  + (v1[2]*v2[0]*v3[1]) - (v1[1]*v2[0]*v3[2]) - (v1[0]*v2[2]*v3[1]) );
60 
61  ADCS_CALC_TYPE gain = 1.0/(ADCS_Add_Tolerance(det_V, 0));
62  invV[0][0] = gain * (v2[1]*v3[2] - v2[2]*v3[1]);
63  invV[0][1] = gain * (v2[2]*v3[0] - v2[0]*v3[2]);
64  invV[0][2] = gain * (v2[0]*v3[1] - v2[1]*v3[0]);
65  invV[1][0] = gain * (v1[2]*v3[1] - v1[1]*v3[2]);
66  invV[1][1] = gain * (v1[0]*v3[2] - v1[2]*v3[0]);
67  invV[1][2] = gain * (v1[1]*v3[0] - v1[0]*v3[1]);
68  invV[2][0] = gain * (v1[1]*v2[2] - v1[2]*v2[1]);
69  invV[2][1] = gain * (v1[2]*v2[0] - v1[0]*v2[2]);
70  invV[2][2] = gain * (v1[0]*v2[1] - v1[1]*v2[0]);
71 
72  // Euler angles matrix (matrices multiplication)
73  for(uint8_t i = 0; i < ADCS_NUM_AXES; i++) {
74  for(uint8_t j = 0; j < ADCS_NUM_AXES; j++) {
75  eulerAnglesMatrix[i][j] = 0;
76  for(uint8_t k = 0; k < ADCS_NUM_AXES; k++) {
77  eulerAnglesMatrix[i][j] += (measMaxtrix[i][k] * invV[k][j]);
78  }
79  }
80  }
81 }
