14void ADS_Measurement_Hybrid(const ADCS_CALC_TYPE v_1[ADCS_NUM_AXES], const ADCS_CALC_TYPE v_2[ADCS_NUM_AXES], const ADCS_CALC_TYPE m_1[ADCS_NUM_AXES],
15 const ADCS_CALC_TYPE m_2[ADCS_NUM_AXES], ADCS_CALC_TYPE eulerAnglesMatrix[ADCS_NUM_AXES][ADCS_NUM_AXES]) {
16
17 // Obtain both ephemeris and measurements vectors
18 ADCS_CALC_TYPE m1[ADCS_NUM_AXES];
19 memcpy(m1, m_1, ADCS_NUM_AXES*sizeof(ADCS_CALC_TYPE));
20 ADCS_CALC_TYPE v1[ADCS_NUM_AXES];
21 memcpy(v1, v_1, ADCS_NUM_AXES*sizeof(ADCS_CALC_TYPE));
22
23 ADCS_CALC_TYPE m1_norm = ADCS_Add_Tolerance(ADCS_VectorNorm(m1),0);
24 ADCS_CALC_TYPE v1_norm = ADCS_Add_Tolerance(ADCS_VectorNorm(v1),0);
25 for(uint8_t i = 0; i < ADCS_NUM_AXES; i++) {
26 m1[i] *= (1.0/m1_norm);
27 v1[i] *= (1.0/v1_norm);
28 }
29
30 ADCS_CALC_TYPE m2[ADCS_NUM_AXES];
31 m2[0] = m_1[1]*m_2[2]-m_1[2]*m_2[1];
32 m2[1] = m_1[2]*m_2[0]-m_1[0]*m_2[2];
33 m2[2] = m_1[0]*m_2[1]-m_1[1]*m_2[0];
34
35 ADCS_CALC_TYPE v2[ADCS_NUM_AXES];
36 v2[0] = v_1[1]*v_2[2]-v_1[2]*v_2[1];
37 v2[1] = v_1[2]*v_2[0]-v_1[0]*v_2[2];
38 v2[2] = v_1[0]*v_2[1]-v_1[1]*v_2[0];
39
40 ADCS_CALC_TYPE m2_norm = ADCS_Add_Tolerance(ADCS_VectorNorm(m2),0);
41 ADCS_CALC_TYPE v2_norm = ADCS_Add_Tolerance(ADCS_VectorNorm(v2),0);
42 for(uint8_t i = 0; i < ADCS_NUM_AXES; i++) {
43 m2[i] *= (1.0/m2_norm);
44 v2[i] *= (1.0/v2_norm);
45 }
46
47 ADCS_CALC_TYPE m3[ADCS_NUM_AXES];
48 m3[0] = m_1[1]*m2[2]-m_1[2]*m2[1];
49 m3[1] = m_1[2]*m2[0]-m_1[0]*m2[2];
50 m3[2] = m_1[0]*m2[1]-m_1[1]*m2[0];
51
52 ADCS_CALC_TYPE v3[ADCS_NUM_AXES];
53 v3[0] = v_1[1]*v2[2]-v_1[2]*v2[1];
54 v3[1] = v_1[2]*v2[0]-v_1[0]*v2[2];
55 v3[2] = v_1[0]*v2[1]-v_1[1]*v2[0];
56
57 ADCS_CALC_TYPE m3_norm = ADCS_Add_Tolerance(ADCS_VectorNorm(m3),0);
58 ADCS_CALC_TYPE v3_norm = ADCS_Add_Tolerance(ADCS_VectorNorm(v3),0);
59 for(uint8_t i = 0; i < ADCS_NUM_AXES; i++) {
60 m3[i] *= (1.0/m3_norm);
61 v3[i] *= (1.0/v3_norm);
62 }
63
64 // Measurements and ephemerides matrix
65 ADCS_CALC_TYPE measMaxtrix[ADCS_NUM_AXES][ADCS_NUM_AXES];
66 ADCS_CALC_TYPE epheMatrix[ADCS_NUM_AXES][ADCS_NUM_AXES];
67 for(uint8_t i = 0; i < ADCS_NUM_AXES; i++) {
68 measMaxtrix[i][0] = m1[i];
69 measMaxtrix[i][1] = m2[i];
70 measMaxtrix[i][2] = m3[i];
71 epheMatrix[i][0] = v1[i];
72 epheMatrix[i][1] = v2[i];
73 epheMatrix[i][2] = v3[i];
74 }
75
76 // Transpose of the ephemerides matrix
77 ADCS_CALC_TYPE invEpheMatrix[ADCS_NUM_AXES][ADCS_NUM_AXES];
78 for(uint8_t i = 0; i < ADCS_NUM_AXES; i++) {
79 for(uint8_t j = 0; j < ADCS_NUM_AXES; j++) {
80 invEpheMatrix[i][j] = epheMatrix[j][i];
81 }
82 }
83
84 // Euler angles matrix (matrices multiplication)
85 for(uint8_t i = 0; i < ADCS_NUM_AXES; i++) {
86 for(uint8_t j = 0; j < ADCS_NUM_AXES; j++) {
87 eulerAnglesMatrix[i][j] = 0;
88 for(uint8_t k = 0; k < ADCS_NUM_AXES; k++) {
89 eulerAnglesMatrix[i][j] += (measMaxtrix[i][k] * invEpheMatrix[k][j]);
90 }
91 }
92 }
93
94}
