14 /*********************** Main function ***********************/
15 void ACS_BdotFunction(const ADCS_CALC_TYPE omega[ADCS_NUM_AXES], const ADCS_CALC_TYPE mag[ADCS_NUM_AXES], const ADCS_CALC_TYPE coilChar[ADCS_NUM_AXES][ADCS_NUM_AXES],
16  const ADCS_CALC_TYPE meanOrbitalMotion, const ADCS_CALC_TYPE orbInclination, const ADCS_CALC_TYPE minInertialMoment,
17  ADCS_CALC_TYPE intensity[ADCS_NUM_AXES]) {
18  // Constants definition
19  // Module of the magnetic field intensity
20  const ADCS_CALC_TYPE B_module = ADCS_Add_Tolerance(ADCS_VectorNorm(mag), 0);
21  FOSSASAT_DEBUG_PRINT(F("B_module = \t"));
22  FOSSASAT_DEBUG_PRINTLN(B_module, 4);
23 
24  // Controller gain constant
25  const ADCS_CALC_TYPE gainConstant = 2.0*meanOrbitalMotion*(1.0 + sin(orbInclination)) * minInertialMoment;
26  FOSSASAT_DEBUG_PRINT(F("gainConstant = "));
27  FOSSASAT_DEBUG_PRINTLN(gainConstant, 4);
28 
29  // General gain
30  const ADCS_CALC_TYPE gainGeneral = gainConstant/pow(B_module, 2);
31  FOSSASAT_DEBUG_PRINT(F("gainGeneral = \t"));
32  FOSSASAT_DEBUG_PRINTLN(gainGeneral, 4);
33 
34  // Generate the needed magnetic moment by means of a vector product: m = k*(omega x b) = (K/B_module^2)*(omega x B)
35  ADCS_CALC_TYPE magMoment[ADCS_NUM_AXES];
36  magMoment[0] = gainGeneral*(omega[1]*mag[2] - omega[2]*mag[1]);
37  magMoment[1] = gainGeneral*(omega[2]*mag[0] - omega[0]*mag[2]);
38  magMoment[2] = gainGeneral*(omega[0]*mag[1] - omega[1]*mag[0]);
39  FOSSASAT_DEBUG_PRINT_ADCS_VECTOR(magMoment, ADCS_NUM_AXES);
40 
41  // Definition of intensity output -solving the equation: A*I = m-
42  for(uint8_t i = 0; i < ADCS_NUM_AXES; i++) {
43  intensity[i] = 0;
44  for(uint8_t j = 0; j < ADCS_NUM_AXES; j++) {
45  intensity[i] += coilChar[i][j] * magMoment[j];
46  }
47  }
48 }
