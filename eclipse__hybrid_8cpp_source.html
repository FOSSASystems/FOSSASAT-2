15 void ADS_Eclipse_Hybrid(const ADCS_CALC_TYPE magData[ADCS_NUM_AXES], const ADCS_CALC_TYPE magEphe[ADCS_NUM_AXES],
16  ADCS_CALC_TYPE rotationMatrix[ADCS_NUM_AXES][ADCS_NUM_AXES]) {
17  // Variables declaration
18  ADCS_CALC_TYPE bodyFrame[ADCS_NUM_AXES];
19  ADCS_CALC_TYPE lvFrame[ADCS_NUM_AXES];
20 
21  // Unitary frame vectors
22  const ADCS_CALC_TYPE dataGain = 1.0 / ADCS_Add_Tolerance(ADCS_VectorNorm(magData), 0);
23  const ADCS_CALC_TYPE epheGain = 1.0 / ADCS_Add_Tolerance(ADCS_VectorNorm(magEphe), 0);
24 
25  // Unitary vector calculation in the body frame
26  bodyFrame[0] = dataGain * magData[0];
27  bodyFrame[1] = dataGain * magData[1];
28  bodyFrame[2] = dataGain * magData[2];
29 
30  // Unitary vector calculation in the LV frame
31  lvFrame[0] = epheGain * magEphe[0];
32  lvFrame[1] = epheGain * magEphe[1];
33  lvFrame[2] = epheGain * magEphe[2];
34 
35  // Dot product between the two (cos(alpha) == a dot b)
36  ADCS_CALC_TYPE alphaDot = bodyFrame[0]*lvFrame[0]+bodyFrame[1]*lvFrame[1]+bodyFrame[2]*lvFrame[2];
37 
38  // Cross product between the two (sin(alpha) == norm(a cross b))
39  ADCS_CALC_TYPE rotationAxe[ADCS_NUM_AXES];
40  rotationAxe[0] = bodyFrame[2]*lvFrame[1]-bodyFrame[1]*lvFrame[2];
41  rotationAxe[1] = bodyFrame[0]*lvFrame[2]-bodyFrame[2]*lvFrame[0];
42  rotationAxe[2] = bodyFrame[1]*lvFrame[0]-bodyFrame[0]*lvFrame[1];
43 
44  ADCS_CALC_TYPE alphaCross = sqrt(pow(rotationAxe[0],2)+pow(rotationAxe[1],2)+pow(rotationAxe[2],2));
45 
46  // Euler angles matrix -Rodrigues' formula-
47  ADCS_CALC_TYPE rotationMatrixAux1[ADCS_NUM_AXES][ADCS_NUM_AXES] = {{alphaDot, 0, 0},
48  {0, alphaDot, 0},
49  {0, 0, alphaDot}};
50  ADCS_CALC_TYPE rotationMatrixAux2[ADCS_NUM_AXES][ADCS_NUM_AXES];
51  for(uint32_t i = 0; i < ADCS_NUM_AXES; i++){
52  for(uint32_t j = 0; j < ADCS_NUM_AXES; j++){
53  rotationMatrixAux2[i][j] = (1.0-alphaDot)*(rotationAxe[i]*rotationAxe[j]);
54  }
55  }
56 
57  ADCS_CALC_TYPE rotationMatrixAux3[ADCS_NUM_AXES][ADCS_NUM_AXES] = {{0, -1.0*rotationAxe[2], rotationAxe[1]},
58  {rotationAxe[2], 0, -1.0*rotationAxe[0]},
59  {-1.0*rotationAxe[1], rotationAxe[0], 0}};
60  for(uint32_t i = 0; i < ADCS_NUM_AXES; i++){
61  for(uint32_t j = 0; j < ADCS_NUM_AXES; j++){
62  rotationMatrixAux3[i][j] *= -1.0*alphaCross;
63  }
64  }
65 
66  // Final computation
67  for(uint32_t i = 0; i < ADCS_NUM_AXES; i++){
68  for(uint32_t j = 0; j < ADCS_NUM_AXES; j++){
69  rotationMatrix[i][j] = rotationMatrixAux1[i][j]+rotationMatrixAux2[i][j]+rotationMatrixAux3[i][j];
70  }
71  }
72 
73 }
