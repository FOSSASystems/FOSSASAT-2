55 static inline int mod255(int x)
56 {
57  while(x >= 255)
58  {
59  x -= 255;
60  x = (x >> 8) + (x & 255);
61  }
62  return(x);
63 }
64 #define MODNN(x) mod255(x)

76 /* Portable C version */
77 void encode_rs_8(uint8_t *data, uint8_t *parity, int pad)
78 {
79  int i, j;
80 
81  // cppcheck-suppress variableScope
82  uint8_t feedback;
83 
84  memset(parity, 0, NROOTS * sizeof(uint8_t));
85 
86  for(i = 0; i < NN - NROOTS - pad; i++)
87  {
88  feedback = INDEX_OF[data[i] ^ parity[0]];
89  if(feedback != A0) /* feedback term is non-zero */
90  {
91  for(j = 1; j < NROOTS; j++)
92  parity[j] ^= ALPHA_TO[mod255(feedback + GENPOLY[NROOTS - j])];
93  }
94 
95  /* Shift */
96  memmove(&parity[0], &parity[1], sizeof(uint8_t) * (NROOTS - 1));
97  if(feedback != A0)
98  parity[NROOTS - 1] = ALPHA_TO[mod255(feedback + GENPOLY[0])];
99  else
100  parity[NROOTS - 1] = 0;
101  }
102 }

104 // cppcheck-suppress unusedFunction
105 int decode_rs_8(uint8_t *data, int *eras_pos, int no_eras, int pad)
106 {
107  int deg_lambda, el, deg_omega;
108  int i, j, r, k;
109  uint8_t u, q, tmp, num1, num2, den, discr_r;
110  uint8_t lambda[NROOTS + 1], s[NROOTS]; /* Err+Eras Locator poly
111  * and syndrome poly */
112  uint8_t b[NROOTS + 1], t[NROOTS + 1], omega[NROOTS + 1];
113  uint8_t root[NROOTS], reg[NROOTS + 1], loc[NROOTS];
114  int syn_error, count;
115 
116  if(pad < 0 || pad > 222) return(-1);
117 
118  /* form the syndromes; i.e., evaluate data(x) at roots of g(x) */
119  for(i = 0; i < NROOTS; i++) s[i] = data[0];
120 
121  for(j = 1; j < NN - pad; j++)
122  {
123  for(i = 0; i < NROOTS; i++)
124  {
125  if(s[i] == 0) s[i] = data[j];
126  else s[i] = data[j] ^ ALPHA_TO[MODNN(INDEX_OF[s[i]] + (FCR + i) * PRIM)];
127  }
128  }
129 
130  /* Convert syndromes to index form, checking for nonzero condition */
131  syn_error = 0;
132  for(i = 0; i < NROOTS; i++)
133  {
134  syn_error |= s[i];
135  s[i] = INDEX_OF[s[i]];
136  }
137 
138  if(!syn_error)
139  {
140  /* if syndrome is zero, data[] is a codeword and there are no
141  * errors to correct. So return data[] unmodified
142  */
143  count = 0;
144  goto finish;
145  }
146 
147  memset(&lambda[1], 0, NROOTS * sizeof(lambda[0]));
148  lambda[0] = 1;
149 
150  if(no_eras > 0)
151  {
152  /* Init lambda to be the erasure locator polynomial */
153  lambda[1] = ALPHA_TO[MODNN(PRIM * (NN - 1 - eras_pos[0]))];
154  for(i = 1; i < no_eras; i++)
155  {
156  u = MODNN(PRIM * (NN - 1 - eras_pos[i]));
157  for(j = i + 1; j > 0; j--)
158  {
159  tmp = INDEX_OF[lambda[j - 1]];
160  if(tmp != A0) lambda[j] ^= ALPHA_TO[MODNN(u + tmp)];
161  }
162  }
163 
164  }
165 
166  for(i = 0; i < NROOTS + 1; i++)
167  b[i] = INDEX_OF[lambda[i]];
168 
169  /*
170  * Begin Berlekamp-Massey algorithm to determine error+erasure
171  * locator polynomial
172  */
173  r = no_eras;
174  el = no_eras;
175  while(++r <= NROOTS) /* r is the step number */
176  {
177  /* Compute discrepancy at the r-th step in poly-form */
178  discr_r = 0;
179  for(i = 0; i < r; i++)
180  {
181  if((lambda[i] != 0) && (s[r - i - 1] != A0))
182  {
183  discr_r ^= ALPHA_TO[MODNN(INDEX_OF[lambda[i]] + s[r - i - 1])];
184  }
185  }
186  discr_r = INDEX_OF[discr_r]; /* Index form */
187  if(discr_r == A0)
188  {
189  /* 2 lines below: B(x) <-- x*B(x) */
190  memmove(&b[1], b, NROOTS * sizeof(b[0]));
191  b[0] = A0;
192  }
193  else
194  {
195  /* 7 lines below: T(x) <-- lambda(x) - discr_r*x*b(x) */
196  t[0] = lambda[0];
197  for(i = 0; i < NROOTS; i++)
198  {
199  if(b[i] != A0)
200  t[i + 1] = lambda[i + 1] ^ ALPHA_TO[MODNN(discr_r + b[i])];
201  else
202  t[i + 1] = lambda[i + 1];
203  }
204 
205  if(2 * el <= r + no_eras - 1)
206  {
207  el = r + no_eras - el;
208  /*
209  * 2 lines below: B(x) <-- inv(discr_r) *
210  * lambda(x)
211  */
212  for(i = 0; i <= NROOTS; i++)
213  b[i] = (lambda[i] == 0) ? A0 : MODNN(INDEX_OF[lambda[i]] - discr_r + NN);
214  }
215  else
216  {
217  /* 2 lines below: B(x) <-- x*B(x) */
218  memmove(&b[1], b, NROOTS * sizeof(b[0]));
219  b[0] = A0;
220  }
221 
222  memcpy(lambda, t, (NROOTS + 1) * sizeof(t[0]));
223  }
224  }
225 
226  /* Convert lambda to index form and compute deg(lambda(x)) */
227  deg_lambda = 0;
228  for(i = 0; i < NROOTS + 1; i++)
229  {
230  lambda[i] = INDEX_OF[lambda[i]];
231  if(lambda[i] != A0) deg_lambda = i;
232  }
233 
234  /* Find roots of the error+erasure locator polynomial by Chien search */
235  memcpy(&reg[1], &lambda[1], NROOTS * sizeof(reg[0]));
236  count = 0; /* Number of roots of lambda(x) */
237  for(i = 1, k = IPRIM - 1; i <= NN; i++, k = MODNN(k + IPRIM))
238  {
239  q = 1; /* lambda[0] is always 0 */
240  for(j = deg_lambda; j > 0; j--)
241  {
242  if(reg[j] != A0)
243  {
244  reg[j] = MODNN(reg[j] + j);
245  q ^= ALPHA_TO[reg[j]];
246  }
247  }
248 
249  if (q != 0) continue; /* Not a root */
250 
251  /* store root (index-form) and error location number */
252  root[count] = i;
253  loc[count] = k;
254  /* If we've already found max possible roots,
255  * abort the search to save time
256  */
257  if(++count == deg_lambda) break;
258  }
259 
260  if(deg_lambda != count)
261  {
262  /*
263  * deg(lambda) unequal to number of roots => uncorrectable
264  * error detected
265  */
266  count = -1;
267  goto finish;
268  }
269 
270  /*
271  * Compute err+eras evaluator poly omega(x) = s(x)*lambda(x) (modulo
272  * x**NROOTS). in index form. Also find deg(omega).
273  */
274  deg_omega = deg_lambda - 1;
275  for(i = 0; i <= deg_omega; i++)
276  {
277  tmp = 0;
278  for(j = i; j >= 0; j--)
279  {
280  if((s[i - j] != A0) && (lambda[j] != A0))
281  tmp ^= ALPHA_TO[MODNN(s[i - j] + lambda[j])];
282  }
283  omega[i] = INDEX_OF[tmp];
284  }
285 
286  /*
287  * Compute error values in poly-form. num1 = omega(inv(X(l))), num2 =
288  * inv(X(l))**(FCR-1) and den = lambda_pr(inv(X(l))) all in poly-form
289  */
290  for(j = count - 1; j >= 0; j--)
291  {
292  num1 = 0;
293  for(i = deg_omega; i >= 0; i--)
294  {
295  if(omega[i] != A0) num1 ^= ALPHA_TO[MODNN(omega[i] + i * root[j])];
296  }
297  num2 = ALPHA_TO[MODNN(root[j] * (FCR - 1) + NN)];
298  den = 0;
299 
300  /* lambda[i+1] for i even is the formal derivative lambda_pr of lambda[i] */
301  for(i = MIN(deg_lambda, NROOTS - 1) & ~1; i >= 0; i -= 2)
302  {
303  if(lambda[i + 1] != A0) den ^= ALPHA_TO[MODNN(lambda[i + 1] + i * root[j])];
304  }
305 
306  /* Apply error to data */
307  if(num1 != 0 && loc[j] >= pad)
308  {
309  data[loc[j] - pad] ^= ALPHA_TO[MODNN(INDEX_OF[num1] + INDEX_OF[num2] + NN - INDEX_OF[den])];
310  }
311  }
312 
313 finish:
314  if(eras_pos != NULL)
315  {
316  for(i = 0; i < count; i++) eras_pos[i] = loc[i];
317  }
318 
319  return(count);
320 }
